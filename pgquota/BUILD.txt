Run the following in the root of the git checkout:

export VERSION=$(grep "Version" pgquota/pgquota.spec | cut -f 2)

mkdir -p ~/rmpbuild/SOURCES
mkdir /tmp/pgquota-${VERSION}
cp pgquota/pgquota /tmp/pgquota-${VERSION}
tar czf ~/rpmbuild/SOURCES/gquota-${VERSION}.tar.gz /tmp/pgquota-${VERSION}
rm -rf /tmp/pgquota-${VERSION}

rpmbuild -ba pgquota/pgquota.spec

# resulting RPM can be found in ~/rpmbuild/RPMS/noarch/
